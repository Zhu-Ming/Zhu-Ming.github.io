题目：

本题要求编写程序，计算 2 个有理数的和、差、积、商。

输入格式：
输入在一行中按照 a1/b1 a2/b2 的格式给出两个分数形式的有理数，其中分子和分母全是整型范围内的整数，负号只可能出现在分子前，分母不为 0。

输出格式：
分别在 4 行中按照 有理数1 运算符 有理数2 = 结果 的格式顺序输出 2 个有理数的和、差、积、商。注意输出的每个有理数必须是该有理数的最简形式 k a/b，其中 k 是整数部分，a/b 是最简分数部分；若为负数，则须加括号；若除法分母为 0，则输出 Inf。题目保证正确的输出中没有超过整型范围的整数。

输入样例 1：

2/3 -4/2

输出样例 1：

2/3 + (-2) = (-1 1/3)
2/3 - (-2) = 2 2/3
2/3 * (-2) = (-1 1/3)
2/3 / (-2) = (-1/3)

输入样例 2：

5/3 0/6

输出样例 2：

1 2/3 + 0 = 1 2/3
1 2/3 - 0 = 1 2/3
1 2/3 * 0 = 0
1 2/3 / 0 = Inf

分析：

输入：按格式输入两个有理数（分母一定不为0，符号一定在分子上，分子可能为0）.
处理：首先定义一个求最大公约数的函数（注意求最大公约数时的算法一定要快，而且注意负数与正数之间也是有最大公约数的，其最大公约数是正数）。然后定义一个符合题目格式的函数用于输出运算后的结果（先对每个结果判断分母是否为0，如果分母为0直接输出Inf,如果不为0，再判断分子是否为0,为0就直接输出0，否则再对其先进行约分{注意一定要定义一个变量保存最大公约数，因为后面再约分时数字已经改变}，然后在根据分子是否时是负数对其进行假分数的相关操作，），再定义一个求加，减，乘，除的函数。注意输入的所有数字一定要是long型的不然会有俩个测试点过不了。
输出：。。。

代码：

#include<stdio.h>
long gcd(long a,long b)//最大公约数
{
    if(a<0){
        a=-a;
    }
    if(b<0){
        b=-b;
    }
    long n=(a>b?a:b),m=(a>b?b:a);
    long c=a%b;
    while(c){
        a=b;
        b=c;
        c=a%b;
    }
    return b;
}
void form(long a,long b){
    if(b==0){
        printf("Inf");
        return;
    }
    if(a==0){
        printf("0");
        return;
    }
    long n=gcd(a,b);//对其进行约分
    a=a/n;
    b=b/n;
    if(a>0){
        long k=a/b;
        if(k==0){
            printf("%ld/%ld",a,b);
            return;
        }else{
            a=a%b;
            if(a==0){
                printf("%ld",k);
                return;
            }else{
                printf("%ld %ld/%ld",k,a,b);
                return;
            }
        }
    }else{
        printf("(");
        long k=a/b;
        if(k==0){
            printf("%ld/%ld)",a,b);
            return;
        }else{
            a=-a;
            a=a%b;
            if(a==0){
                printf("%ld)",k);
                return;
            }else{
                printf("%ld %ld/%ld)",k,a,b);
                return;
            }
        }
    }
}
void add(long a1,long a2,long b1,long b2){
    long k=a2;
    long c2=b2*a2;
    a1=a1*b2;
    a2=a2*b2;
    b1=b1*a2;
    b2=b2*k;
    long c1=a1+b1;
    form(a1,a2);
    printf(" + ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
void ride(long a1,long a2,long b1,long b2){
    long c1=a1*b1;
    long c2=a2*b2;
    form(a1,a2);
    printf(" * ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
void div(long a1,long a2,long b1,long b2){
    long c1=a1*b2;
    long c2=a2*b1;
    if(c2<0){
        c2=-c2;
        c1=-c1;
    }
    form(a1,a2);
    printf(" / ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}

void sub(long a1,long a2,long b1,long b2){
    long k=a2;
    long c2=b2*a2;
    a1=a1*b2;
    b1=b1*a2;
    a2=a2*b2;
    b2=b2*k;
    long c1=a1-b1;
    form(a1,a2);
    printf(" - ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
int main(void){
    long a1,a2,b1,b2;
    scanf("%ld/%ld %ld/%ld",&a1,&a2,&b1,&b2);
    add(a1,a2,b1,b2);
    printf("\n");
    sub(a1,a2,b1,b2);
    printf("\n");
    ride(a1,a2,b1,b2);
    printf("\n");
    div(a1,a2,b1,b2);
    return 0;
}
