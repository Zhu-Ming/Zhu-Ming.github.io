PTA-Basic-1034

2019-09-22
PTA

题目：

本题要求编写程序，计算 2 个有理数的和、差、积、商。

输入格式：
输入在一行中按照 a1/b1 a2/b2 的格式给出两个分数形式的有理数，其中分子和分母全是整型范围内的整数，负号只可能出现在分子前，分母不为 0。

输出格式：
分别在 4 行中按照 有理数1 运算符 有理数2 = 结果 的格式顺序输出 2 个有理数的和、差、积、商。注意输出的每个有理数必须是该有理数的最简形式 k a/b，其中 k 是整数部分，a/b 是最简分数部分；若为负数，则须加括号；若除法分母为 0，则输出 Inf。题目保证正确的输出中没有超过整型范围的整数。

输入样例 1：

2/3 -4/2

输出样例 1：

2/3 + (-2) = (-1 1/3)
2/3 - (-2) = 2 2/3
2/3 * (-2) = (-1 1/3)
2/3 / (-2) = (-1/3)

输入样例 2：

5/3 0/6

输出样例 2：

1 2/3 + 0 = 1 2/3
1 2/3 - 0 = 1 2/3
1 2/3 * 0 = 0
1 2/3 / 0 = Inf

分析：

输入：按格式输入两个有理数（分母一定不为0，符号一定在分子上，分子可能为0）.
处理：首先定义一个求最大公约数的函数（注意求最大公约数时的算法一定要快，而且注意负数与正数之间也是有最大公约数的，其最大公约数是正数）。然后定义一个符合题目格式的函数用于输出运算后的结果（先对每个结果判断分母是否为0，如果分母为0直接输出Inf,如果不为0，再判断分子是否为0,为0就直接输出0，否则再对其先进行约分{注意一定要定义一个变量保存最大公约数，因为后面再约分时数字已经改变}，然后在根据分子是否时是负数对其进行假分数的相关操作，），再定义一个求加，减，乘，除的函数。注意输入的所有数字一定要是long型的不然会有俩个测试点过不了。
输出：。。。

代码：

#include<stdio.h>
long gcd(long a,long b)//最大公约数
{
    if(a<0){
        a=-a;
    }
    if(b<0){
        b=-b;
    }
    long n=(a>b?a:b),m=(a>b?b:a);
    long c=a%b;
    while(c){
        a=b;
        b=c;
        c=a%b;
    }
    return b;
}
void form(long a,long b){
    if(b==0){
        printf("Inf");
        return;
    }
    if(a==0){
        printf("0");
        return;
    }
    long n=gcd(a,b);//对其进行约分
    a=a/n;
    b=b/n;
    if(a>0){
        long k=a/b;
        if(k==0){
            printf("%ld/%ld",a,b);
            return;
        }else{
            a=a%b;
            if(a==0){
                printf("%ld",k);
                return;
            }else{
                printf("%ld %ld/%ld",k,a,b);
                return;
            }
        }
    }else{
        printf("(");
        long k=a/b;
        if(k==0){
            printf("%ld/%ld)",a,b);
            return;
        }else{
            a=-a;
            a=a%b;
            if(a==0){
                printf("%ld)",k);
                return;
            }else{
                printf("%ld %ld/%ld)",k,a,b);
                return;
            }
        }
    }
}
void add(long a1,long a2,long b1,long b2){
    long k=a2;
    long c2=b2*a2;
    a1=a1*b2;
    a2=a2*b2;
    b1=b1*a2;
    b2=b2*k;
    long c1=a1+b1;
    form(a1,a2);
    printf(" + ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
void ride(long a1,long a2,long b1,long b2){
    long c1=a1*b1;
    long c2=a2*b2;
    form(a1,a2);
    printf(" * ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
void div(long a1,long a2,long b1,long b2){
    long c1=a1*b2;
    long c2=a2*b1;
    if(c2<0){
        c2=-c2;
        c1=-c1;
    }
    form(a1,a2);
    printf(" / ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}

void sub(long a1,long a2,long b1,long b2){
    long k=a2;
    long c2=b2*a2;
    a1=a1*b2;
    b1=b1*a2;
    a2=a2*b2;
    b2=b2*k;
    long c1=a1-b1;
    form(a1,a2);
    printf(" - ");
    form(b1,b2);
    printf(" = ");
    form(c1,c2);
}
int main(void){
    long a1,a2,b1,b2;
    scanf("%ld/%ld %ld/%ld",&a1,&a2,&b1,&b2);
    add(a1,a2,b1,b2);
    printf("\n");
    sub(a1,a2,b1,b2);
    printf("\n");
    ride(a1,a2,b1,b2);
    printf("\n");
    div(a1,a2,b1,b2);
    return 0;
}

PTA-Basic-1033

2019-09-22
PTA

题目：

旧键盘上坏了几个键，于是在敲一段文字的时候，对应的字符就不会出现。现在给出应该输入的一段文字、以及坏掉的那些键，打出的结果文字会是怎样？

输入格式：
输入在 2 行中分别给出坏掉的那些键、以及应该输入的文字。其中对应英文字母的坏键以大写给出；每段文字是不超过 10
^5
​​ 个字符的串。可用的字符包括字母 [a-z, A-Z]、数字 0-9、以及下划线 _（代表空格）、,、.、-、+（代表上档键）。题目保证第 2 行输入的文字串非空。

注意：如果上档键坏掉了，那么大写的英文字母无法被打出。

输出格式：
在一行中输出能够被打出的结果文字。如果没有一个字符能被打出，则输出空行。

输入样例：

7+IE.
7_This_is_a_test.

输出样例：

_hs_s_a_tst

分析:

输入：第一行输入坏掉的字符（如果有字母则对应
的字母不能出现，如果有下划线则不下划线，有加号大写字母甭能出现）
处理：
先定义两个尽可能大的字符数组一个用来存储坏键，一个用来存储要输入的字符串。

定义一个删除字符的函数，然后遍历第一个字符串分别在第二个字符串中删去相应的字符。

输出：处理后的字符串如果字符串长度为0则输出空行。

代码：

#include<stdio.h>
#include<string.h>
#include<ctype.h>
char * Remove(char*a,char b){
    char * src,*dst;
    src=a;
    dst=a;
    while(*src!='\0'){
        *dst=*src;
        src++;
        if(*dst!=b){
            dst++;
        }
    }
    *dst='\0';
    return a;
}
int main(void){
    char a[110000]={},b[110000]={};
    gets(a);
    gets(b);
    int i=0;
    while(a[i]!='\0'){
                if(a[i]!='+'){//shang dang jian mei hua
                            if(a[i]>='A'&&a[i]<='Z'){
                                Remove(b,tolower(a[i]));
                                Remove(b,toupper(a[i]));
                            }else{
                                Remove(b,a[i]);
                            }
                }else{
                    for(char j='A';j<('Z'+1);j++){
                        Remove(b,j);
                    }
            }
        i++;
    }

    if(strlen(b)!=0){
        printf("%s",b);
    }else{
        printf("\n");
    }
    return 0;
}

PTA-Basic-1032

2019-09-22
PTA

题目：

为了用事实说明挖掘机技术到底哪家强，PAT 组织了一场挖掘机技能大赛。现请你根据比赛结果统计出技术最强的那个学校。

输入格式：
输入在第 1 行给出不超过 10
​5
​​ 的正整数 N，即参赛人数。随后 N 行，每行给出一位参赛者的信息和成绩，包括其所代表的学校的编号（从 1 开始连续编号）、及其比赛成绩（百分制），中间以空格分隔。

输出格式：
在一行中给出总得分最高的学校的编号、及其总分，中间以空格分隔。题目保证答案唯一，没有并列。

输入样例：

6
3 65
2 80
1 100
2 70
3 40
3 0

输出样例：

2 150

分析：

输入：第一行给出要输入的数目，随后每一行给出一个学校的得分。
处理：
先定义一个学校编号的数组越大越好（其数组下标就是输入的学校的编号），定义一个得分最大的学校的编号，与对应的分数，然后输入n次一个学校的编号与得分，如果输入的分数加到对应的学校后比当前最高的学校搞，就改变两个量，最后输入结束时就得到了，最大的学校的编号与分数了。
输出：学校的编号 学校的总分数。
注意：此题如果采用排序的话可能会运行超时。

代码：

#include<stdio.h>
int main(void){
    int schools[100001]={0};
    int max=0;
    int max_school=0;
    int n,a,b;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d %d",&a,&b);
        schools[a]=schools[a]+b;
        if(schools[a]>max){
            max=schools[a];
            max_school=a;
        }
    }
    printf("%d %d",max_school,max);
    return 0;
}
