#1.1.1字符串复制
函数:char * strcpy (char* destination,const char* source);

  • 此函数会将指针source指向的字符串完全复制给destination,包括末尾的’\0’,所以在前面的指针指向的字符串长度一定要大于等于后一个。

此函数的实现：
char * strcpy (char* destination,const char* source){

char* copy_dest=destination;

while(*destination++=*source++);

return copy_dest;

}

#####注意：p++==(p++),当读到’\0’时，返回0。
函数：char * strncpy (char * dest,const char * source,size_t num);

此函数从source中读取指定长度的字符串到dest中。

实现：

char * strncpy (char * dest,const char * src,size_t num){

char * copy_dest=dest;

int i=0;

while(i<num){

i++;

if(*dest++=*src++){

    continue;

}

*dest='\0';        

}

return copy_dest;

}

注意：size_t 是无符号整型，相当于unsigned int,同时使用此函数时也需要主动加入null终结符即使用完函数后要加上（dest[num]=’\0’）,否则打印时会出错。!

2019-11-21

注意此处的串的实现中的模式匹配，并不是最优解法（使用KMP算法）。建议直接放入头文件。此外，但凡程序中涉及的位置皆为位序（从1开始），在某些部分，如果使用坐标作为参数，可能会产生错误，不过基本上这些错误，程序中都有所应对。

#ifndef STRING_H_INCLUDED
#define STRING_H_INCLUDED
#define MAXSIZE 255//默认最大长度设置为255
typedef struct{
    char strs[MAXSIZE];
    int len;
}String;
String* StrInit(String *s){//串的初始化
    s=(String *)malloc(sizeof(String));
    s->len=0;
    return s;
}
void StrAppend(String *s,char ch){//将字符追加到串上
    if(s->len<MAXSIZE){
        s->strs[(s->len)++]=ch;
    }else{
        printf("string is filled\n");
    }
}
void StrAssign(String *s,char *p){//将指向的字符串赋值链接到s后面
    while(*p!='\0'){
        StrAppend(s,*p);
        p++;
    }
}
void StrCopy(String *s1,String *s2){//将s2复制给s1
    if(s2->len==0){
            return;
    }
        s1->len=0;
    for(int i=0;i<(s2->len);i++){
        StrAppend(s1,s2->strs[i]);
    }
}
int StrEmpty(String *s){//判断串是否为空
    if(s==NULL)
        return 1;
    if(s!=NULL)
        return s->len==0;

        printf("must initial\n");
    return 1;
}
int StrLength(String *s){//求串长度
    if(s!=NULL)
    {
        return s->len;
    }
    else
    {
        printf("must initial\n");
        return 0;
    }
}
String * SubString(String *s,int pos,int len){//求子串
    if(pos-1+len>(s->len)){
        printf("the length of the string is over MAXSIZE\n");
        return NULL;
    }else{
        String *temp;
        temp=StrInit(temp);
        for(int i=pos-1;i<len;i++){
            StrAppend(temp,s->strs[i]);
        }
        return temp;
    }
}
String * Conact(String *s1,String *s2){//连接两个字符串，这里并不是最好的解法。
    String *result;
    if(StrLength(s1)+StrLength(s2)>MAXSIZE){
        printf("the length of the string is over MAXSIZE\n");
        result=NULL;
    }else{
        result=StrInit(result);
        for(int i=0;i<StrLength(s1);i++){
            StrAppend(result,s1->strs[i]);
        }

        for(int i=0;i<StrLength(s2);i++){
            StrAppend(result,s2->strs[i]);
        }
    }
        return result;
}
int Index(String *s,int pos,String *sub){//求某个字符串在串中的起始位置
    if((pos+StrLength(sub)>=MAXSIZE)){
        printf("not suitable\n");
        return 0;
    }
    int i=pos-1,j=0;
    while(i<(s->len)&&j<(sub->len)){
        if(s->strs[i]==sub->strs[j]){
            ++i;
            ++j;
        }else{
            i=i-j+1;
            j=0;
        }
    }
    if(j>=sub->len)
        return i-sub->len+1;
    else
        return 0;
}
String * StrInsert(String *s,String *r,int pos){/将r插入到s的第pos个位置处
    String *result;
    if(s->len+r->len>=MAXSIZE){
        printf("the length of the result is over MAXSIZE\n");
        result=NULL;
    }else{
        result=StrInit(result);
        for(int i=0;i<pos-1;i++){
            StrAppend(result,s->strs[i]);
        }
        for(int i=0;i<r->len;i++){
            StrAppend(result,r->strs[i]);
        }
        for(int i=pos-1;i<s->len;i++){
            StrAppend(result,s->strs[i]);
        }
    }
        return result;
}
void ClearString(String *s){//清空串
    s->len=0;
}
String * StrDelete(String *s,int pos,int len){//删除串中指定位置开始，长度为len的字符串。
    String *result;
    result=StrInit(result);
    for(int i=0;i<pos-1;i++){
        StrAppend(result,s->strs[i]);
    }
    for(int i=pos+len-1;i<s->len;i++){
        StrAppend(result,s->strs[i]);
    }
    return result;
}
String * StrReplace(String *s,String *target,String *r){//将s中的所有target子串替换为r
    if(target->len==0){
        return s;
    }
    int index=Index(s,1,target);
    while(index){
        s=StrDelete(s,index,target->len);
        s=StrInsert(s,r,index);
        index=Index(s,1,target);
    }
    return s;
}
void StrPrint(String *s){//打印字符串
    if(s==NULL)
    {
        printf("must initial\n");
    }
    else
    {
        for(int i=0;i<s->len;i++){
            printf("%c",s->strs[i]);
        }
    }
}
void DestroyString(String *s){//销毁字符串
    s->len=0;
    free(s);
}

双链表

2019-11-20

在单链表的基础上添加了一个指向前面结点的指针。

代码：

#include<stdlib.h>
#include<stdio.h>
#define ElementType char
typedef struct DNode{
    ElementType data;
    struct DNode* prior;
    struct DNode* next; 
}DLinkList,DNode;
DLinkList* DList_HeadInsert(DLinkList *s){
    s=(DNode *)malloc(sizeof(DNode));//头结点
    s->prior=NULL;
    s->next=NULL;
    DNode *node;
    ElementType data;
    printf("请输入用于构建单链表的字符（以空格分隔）以换行结尾:");
    data=getchar();
    while(data!='\n'){
        getchar();
        node=(DNode *)malloc(sizeof(DNode));
        node->data=data;
        DNode *nextNode=s->next;
        if(nextNode){
            node->next=nextNode;
            node->prior=nextNode->prior;
            nextNode->prior=node;
            s->next=node;
        }else{
            node->prior=s;
            node->next=s->next;
            s->next=node;
        }
        data=getchar();
    }
    return s;
}
DLinkList* DList_TailInsert(DLinkList *s){
    s=(DLinkList *)malloc(sizeof(DNode));//头结点
    s->next=NULL;
    DNode *r=s;
    DNode *node;
    ElementType data;
    printf("请输入用于构建单链表的字符（以空格分隔）以换行结尾:");
    data=getchar();
    while(data!='\n'){
        getchar();
        node=(DNode *)malloc(sizeof(DNode));
        node->data=data;
        node->next=NULL;
        node->prior=r;
        r->next=node;
        r=node;
        data=getchar();
    }
    return s;
}
int Length(DLinkList *s){
    if(!s)
        return 0;
    int len=0;
    s=s->next;
    while(s){
        len++;
        s=s->next;
    }
    return len;
}
DNode* GetElem(DLinkList *s,int i){
    if(i<0){
        return NULL;
    }
    int curr=0;
    DNode* currNode=s;
    while(currNode&&curr<i){
        curr++;
        currNode=currNode->next;
    }
    return currNode;
}
DNode *LocateElement(DLinkList *l,ElementType e){
    DLinkList *r=l->next;
    while(r&&r->data!=e)r=r->next;
    return r;
}
int isEmpty(DLinkList *l){
    if(l==NULL||l->next==NULL){
        return 1;
    }
    return 0;
}
void DListInsert(DLinkList *s,int pos,ElementType e){
    if(pos<1||pos>Length(s)){
        printf("位置不合法\n");
        return;
    }
    DNode *r=GetElem(s,pos-1);
    DNode *node=(DNode *)malloc(sizeof(DNode));
    node->data=e;
    node->next=r->next;
    r->next->prior=node;
    node->prior=r;
    r->next=node;
}
void DListDelete(DLinkList *l,int pos){
    if(pos<1||pos>Length(l)){
        printf("位置不合法\n");
        return;
    }
    DNode *r=GetElem(l,pos-1);
    DNode *q= r->next; 
    q->next->prior=r;
    r->next=q->next;
    free(q);
}
void PrintDList(DLinkList *l){
    if(!isEmpty(l)){
        int flag=1;
        DNode *temp=l->next;
        while(temp){
            if(flag){
                printf("%c",temp->data);
                flag=0; 
            }else{
                printf("<=>%c",temp->data);
            }
            temp=temp->next;
        }
    }else{
        printf("链表为空\n");
    }
}
void DestroyDList(DLinkList *s){
    free(s);
}
int main(void){
    DLinkList *s=NULL;
    s=DList_TailInsert(s);
    DListInsert(s,0,'x');
    DListDelete(s,2);
    PrintDList(s);
    printf("链表长度是：%d",Length(s));
}

typedef struct LinkNode{

LinkList* List_HeadInsert(LinkList *s){
    s=(LinkList *)malloc(sizeof(struct LinkNode));//头结点
    s->next=NULL;
    LinkList *node;

        node=(LinkList *)malloc(sizeof(struct LinkNode));
        node->data=data;
        node->next=s->next;

LeetCode-28实现strStr()

2019-11-19

题目：

实现 strStr() 函数。

给定一个 haystack 字符串和一个 needle 字符串，在 haystack 字符串中找出 needle 字符串出现的第一个位置 (从0开始)。如果不存在，则返回 -1。

示例 1:

输入: haystack = “hello”, needle = “ll”
输出: 2
示例 2:

输入: haystack = “aaaaa”, needle = “bba”
输出: -1
说明:

当 needle 是空字符串时，我们应当返回什么值呢？这是一个在面试中很好的问题。

对于本题而言，当 needle 是空字符串时我们应当返回 0 。这与C语言的 strstr() 以及 Java的 indexOf() 定义相符。

思路：显然直接调用indexOf()方法的效率是最高的，但是此题考查的是KMP算法，关于这个算法，后续补充。这里就讨巧一下。

代码：

public int strStr(String haystack, String needle) {
    if(needle==null||needle.length()==0)
        return 0;
    return haystack.indexOf(needle);
}
